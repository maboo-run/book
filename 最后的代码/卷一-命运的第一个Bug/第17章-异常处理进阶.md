```python
try:
    # 可能出错的代码
    num = int(input("请输入数字: "))
    result = 100 / num
    print(f"结果: {result}")
except ValueError:
    # 处理值错误（输入不是数字）
    print("输入错误！请输入数字")
except ZeroDivisionError:
    # 处理除零错误
    print("除数不能为零")
except Exception as e:
    # 处理其他所有错误
    print(f"发生未知错误: {e}")
else:
    # 没有异常时执行
    print("计算成功")
finally:
    # 无论如何都会执行
    print("程序结束")
```

"四个部分："

"- `try`: 尝试执行可能出错的代码"
"- `except`: 捕获并处理特定类型的异常"
"- `else`: 如果没有异常，执行这里的代码"
"- `finally`: 无论如何都会执行，常用于清理资源"


```python
def get_number_input(prompt):
    """
    获取用户输入的数字，带异常处理
    
    返回:
        float: 用户输入的数字
        None: 用户选择退出
    
    异常处理:
        - ValueError: 输入无法转换为数字
        - KeyboardInterrupt: 用户按Ctrl+C
    """
    while True:
        try:
            user_input = input(prompt)
            
            # 检查退出
            if user_input.lower() == 'q':
                return None
            
            # 转换为数字
            return float(user_input)
            
        except ValueError:
            print("错误：请输入有效的数字！")
            print("提示：可以输入整数（如10）或小数（如3.14）")
            
        except KeyboardInterrupt:
            print("\n\n检测到退出信号，程序即将结束...")
            return None
        
        except Exception as e:
            # 捕获意外错误
            print(f"发生未知错误: {e}")
            print("请重试或输入'q'退出")
```

```python
def safe_calculate(num1, operator, num2):
    """
    安全的计算函数，带完善的异常处理
    
    返回:
        tuple: (success, result_or_error_msg)
        success为True时，返回(True, 计算结果)
        success为False时，返回(False, 错误信息)
    """
    try:
        if operator == '+':
            result = num1 + num2
        elif operator == '-':
            result = num1 - num2
        elif operator == '*':
            result = num1 * num2
        elif operator == '/':
            if num2 == 0:
                raise ZeroDivisionError("除数不能为零")
            result = num1 / num2
        elif operator == '**':
            # 新增：幂运算
            if num1 == 0 and num2 < 0:
                raise ValueError("0不能进行负数次幂运算")
            result = num1 ** num2
        else:
            raise ValueError(f"不支持的运算符: {operator}")
        
        return (True, result)
        
    except ZeroDivisionError as e:
        return (False, f"数学错误: {e}")
        
    except ValueError as e:
        return (False, f"值错误: {e}")
        
    except OverflowError:
        return (False, "结果太大，超出计算范围")
        
    except Exception as e:
        return (False, f"计算失败: {e}")
```

```python
class CalculatorHistory:
    """计算器历史记录，带异常处理"""
    
    def __init__(self):
        try:
            self.records = []
        except Exception as e:
            print(f"初始化历史记录失败: {e}")
            self.records = []
    
    def add_record(self, num1, operator, num2, result):
        """添加记录，带异常处理"""
        try:
            from datetime import datetime
            
            record = {
                "num1": num1,
                "num2": num2,
                "operator": operator,
                "result": result,
                "time": datetime.now()
            }
            
            self.records.append(record)
            return True
            
        except Exception as e:
            print(f"添加历史记录失败: {e}")
            return False
    
    def show_all(self):
        """显示历史，带异常处理"""
        try:
            if not self.records:
                print("暂无历史记录")
                return
            
            print("\n" + "=" * 60)
            print("  计算历史")
            print("=" * 60)
            
            for i, record in enumerate(self.records, 1):
                time_str = record['time'].strftime("%H:%M:%S")
                expr = f"{record['num1']} {record['operator']} {record['num2']}"
                print(f"{i}. [{time_str}] {expr} = {record['result']}")
            
            print("=" * 60)
            
        except KeyError as e:
            print(f"历史记录数据损坏，缺少字段: {e}")
            
        except Exception as e:
            print(f"显示历史记录失败: {e}")
```

```python
def get_number_input(prompt, max_retries=3):
    """
    获取用户输入，限制重试次数
    """
    retries = 0
    
    while retries < max_retries:
        try:
            user_input = input(prompt)
            
            if user_input.lower() == 'q':
                return None
            
            num = float(user_input)
            
            # 范围检查
            if abs(num) > 1e10:
                print("错误：数字太大，请输入-10000000000到10000000000之间的数")
                retries += 1
                continue
            
            return num
            
        except ValueError:
            retries += 1
            if retries < max_retries:
                print(f"输入错误！还有{max_retries - retries}次机会")
            else:
                print("输入错误次数过多，返回主菜单")
                return None
```

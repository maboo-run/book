```python
def calculate_discount(price, discount_rate, member_level):
    """
    计算折扣后的价格
    
    这个函数根据商品原价、折扣率和会员等级，计算最终价格。
    会员等级越高，额外折扣越多。
    
    参数:
        price (float): 商品原价
        discount_rate (float): 基础折扣率，0-1之间
        member_level (int): 会员等级，1-5级
    
    返回:
        float: 折扣后的价格
    
    示例:
        >>> calculate_discount(100, 0.8, 3)
        72.0
        
    注意:
        - 会员等级必须在1-5之间，否则按1级处理
        - 折扣率必须在0-1之间，否则抛出ValueError
    """
    if not 0 <= discount_rate <= 1:
        raise ValueError("折扣率必须在0-1之间")
    
    # 会员额外折扣：每级额外减5%
    member_discount = 0.95 ** (member_level - 1)
    
    # 计算最终价格
    final_price = price * discount_rate * member_discount
    
    return final_price
```

```python
def calculate_profit_margin(revenue, cost, tax, discount, quantity):
    """计算利润率"""
    gross_income = revenue + discount  # 总收入 = 营收 + 折扣返还
    net_cost = cost - tax  # 净成本 = 成本 - 税收抵扣
    profit = gross_income * net_cost  # 利润
    margin = profit / quantity  # 利润率
    return margin
```

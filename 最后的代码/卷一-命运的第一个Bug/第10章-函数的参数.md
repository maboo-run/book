```python
def calculate_total(price, discount=1, *extras):
    """
    计算商品总价
    
    参数:
        price: 基础价格(必填)
        discount: 折扣(默认1,即不打折)
        *extras: 附加费用(可变参数)
    
    返回:
        总价
    """
    # 计算折后价
    discounted_price = price * discount
    
    # 计算附加费用总和
    total_extras = sum(extras)
    
    # 返回总价
    return discounted_price + total_extras

# 测试
print("只有基础价格:", calculate_total(100))
print("基础价格+折扣:", calculate_total(100, 0.8))
print("基础价格+折扣+附加费:", calculate_total(100, 0.8, 10, 5, 3))
```

```python
def get_number_input(prompt="请输入一个数字:"):
    """
    获取并验证用户的数字输入
    
    参数:

    返回:
        用户输入的数字(float类型)
    """

        user_input = input(prompt)
        try:
            return float(user_input)
        except ValueError:
            print("输入错误!请输入有效的数字")
```

```python
def celsius_to_fahrenheit(celsius):
    """摄氏度转华氏度"""
    return celsius * 9 / 5 + 32

def fahrenheit_to_celsius(fahrenheit):
    """华氏度转摄氏度"""
    return (fahrenheit - 32) * 5 / 9
```

```python
"""
核心计算模块
支持基本和高级数学运算
"""

import math
import logging

logger = logging.getLogger(__name__)

class Calculator:
    """计算器核心引擎"""
    
    # 支持的运算符
    OPERATORS = {
        '+': '加法',
        '-': '减法',
        '*': '乘法',
        '/': '除法',
        '**': '幂运算',
        '%': '取余',
        'sqrt': '开方'
    }
    
    @staticmethod
    def calculate(num1, operator, num2=None):
        """
        执行计算
        
        参数:
            num1: 第一个操作数
            operator: 运算符
            num2: 第二个操作数（某些运算不需要）
        
        返回:
            tuple: (success, result_or_error_message)
        """
        try:
            logger.info(f"计算: {num1} {operator} {num2}")
            
            if operator == '+':
                result = num1 + num2
            elif operator == '-':
                result = num1 - num2
            elif operator == '*':
                result = num1 * num2
            elif operator == '/':
                if num2 == 0:
                    raise ZeroDivisionError("除数不能为零")
                result = num1 / num2
            elif operator == '**':
                if num1 == 0 and num2 < 0:
                    raise ValueError("0不能进行负数次幂运算")
                result = num1 ** num2
            elif operator == '%':
                if num2 == 0:
                    raise ZeroDivisionError("除数不能为零")
                result = num1 % num2
            elif operator == 'sqrt':
                if num1 < 0:
                    raise ValueError("负数不能开方")
                result = math.sqrt(num1)
            else:
                raise ValueError(f"不支持的运算符: {operator}")
            
            logger.info(f"计算成功: {result}")
            return (True, result)
            
        except ZeroDivisionError as e:
            logger.error(f"除零错误: {e}")
            return (False, f"数学错误: {e}")
        except ValueError as e:
            logger.error(f"值错误: {e}")
            return (False, f"值错误: {e}")
        except OverflowError:
            logger.error("结果溢出")
            return (False, "结果太大，超出计算范围")
        except Exception as e:
            logger.error(f"未知错误: {e}", exc_info=True)
            return (False, f"计算失败: {e}")
    
    @staticmethod
    def validate_number(value, max_value=1e10):
        """
        验证数字范围
        
        返回:
            tuple: (valid, number_or_error)
        """
        try:
            num = float(value)
            if abs(num) > max_value:
                return (False, f"数字太大，请输入小于{max_value}的数")
            return (True, num)
        except ValueError:
            return (False, "请输入有效的数字")
```

```python
"""
日志工具
统一的日志配置
"""

import logging
from config.settings import LOG_FILE, LOG_LEVEL, LOG_FORMAT

def setup_logger(name=None):
    """
    配置日志
    
    参数:
        name: 日志器名称
    
    返回:
        Logger对象
    """
    logger = logging.getLogger(name)
    
    # 避免重复添加handler
    if logger.handlers:
        return logger
    
    logger.setLevel(getattr(logging, LOG_LEVEL))
    
    # 文件handler
    file_handler = logging.FileHandler(LOG_FILE, encoding='utf-8')
    file_handler.setLevel(logging.DEBUG)
    file_formatter = logging.Formatter(LOG_FORMAT)
    file_handler.setFormatter(file_formatter)
    
    # 控制台handler
    console_handler = logging.StreamHandler()
    console_handler.setLevel(logging.WARNING)  # 只显示警告和错误
    console_handler.setFormatter(file_formatter)
    
    logger.addHandler(file_handler)
    logger.addHandler(console_handler)
    
    return logger
```

```python
"""
格式化工具
支持彩色输出
"""

from config.settings import USE_COLOR

class Colors:
    """终端颜色代码"""
    RESET = '\033[0m'
    RED = '\033[91m'
    GREEN = '\033[92m'
    YELLOW = '\033[93m'
    BLUE = '\033[94m'
    MAGENTA = '\033[95m'
    CYAN = '\033[96m'

def colored(text, color):
    """
    给文本添加颜色
    
    参数:
        text: 文本内容
        color: 颜色（Colors类的属性）
    """
    if USE_COLOR:
        return f"{color}{text}{Colors.RESET}"
    return text

def success(text):
    """成功消息（绿色）"""
    return colored(f"✓ {text}", Colors.GREEN)

def error(text):
    """错误消息（红色）"""
    return colored(f"✗ {text}", Colors.RED)

def info(text):
    """信息消息（蓝色）"""
    return colored(text, Colors.BLUE)

def warning(text):
    """警告消息（黄色）"""
    return colored(f"⚠ {text}", Colors.YELLOW)
```

```python
def filter_by_operator(self, operator):
    """按运算符筛选"""
    return [r for r in self.records if r['operator'] == operator]

def filter_by_time_range(self, start, end):
    """按时间范围筛选"""
    return [r for r in self.records if start <= r['time'] <= end]

def get_statistics(self):
    """获取统计信息"""
    from collections import Counter
    operators = [r['operator'] for r in self.records]
    return {
        'total': len(self.records),
        'operators': dict(Counter(operators))
    }
```

```python
"""
超级计算器 v2.0
一个专业的命令行计算器
"""

from core.calculator import Calculator
from data.history import HistoryManager
from data.storage import Storage
from ui.interface import UserInterface
from utils.logger import setup_logger

# 配置日志
logger = setup_logger(__name__)

def main():
    """主函数"""
    logger.info("程序启动")
    
    try:
        # 初始化组件
        storage = Storage()
        history = HistoryManager(storage)
        calc = Calculator()
        ui = UserInterface(calc, history)
```

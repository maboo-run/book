from datetime import datetime

history = []

def add_to_history(num1, operator, num2, result):
    """
    添加计算记录到历史
    
    使用字典存储，包含详细信息
    """
    record = {
        "num1": num1,
        "num2": num2,
        "operator": operator,
        "result": result,
        "time": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    }
    history.append(record)

def show_history():
    """显示历史记录"""
    if not history:
        print("暂无历史记录")
        return
    
    print("\n" + "=" * 60)
    print("  计算历史")
    print("=" * 60)
    for i, record in enumerate(history, 1):
        print(f"{i}. [{record['time']}] "
              f"{record['num1']} {record['operator']} {record['num2']} = {record['result']}")
    print("=" * 60)

def find_by_operator(operator):
    """
    按运算符筛选历史记录
    """
    results = [r for r in history if r['operator'] == operator]
    return results

# 测试

```python
from datetime import datetime
from collections import Counter

class CalculatorHistory:
    """计算器历史记录管理"""
    
    def __init__(self):
        self.records = []  # 所有记录
    
    def add_record(self, num1, operator, num2, result):
        """添加一条记录"""
        record = {
            "num1": num1,
            "num2": num2,
            "operator": operator,
            "result": result,
            "time": datetime.now(),
            "expression": f"{num1} {operator} {num2}"
        }
        self.records.append(record)
    
    def get_all(self):
        """获取所有记录"""
        return self.records
    
    def filter_by_operator(self, operator):
        """按运算符筛选"""
        return [r for r in self.records if r['operator'] == operator]
    
    def filter_by_time_range(self, start_time, end_time):
        """按时间范围筛选"""
        return [r for r in self.records 
                if start_time <= r['time'] <= end_time]
    
    def get_statistics(self):
        """统计各类运算次数"""
        operators = [r['operator'] for r in self.records]
        return dict(Counter(operators))
    
    def show_all(self):
        """显示所有记录"""
        if not self.records:
            print("暂无历史记录")
            return
        
        print("\n" + "=" * 70)
        print("  计算历史")
        print("=" * 70)
        for i, record in enumerate(self.records, 1):
            time_str = record['time'].strftime("%Y-%m-%d %H:%M:%S")
            print(f"{i}. [{time_str}] {record['expression']} = {record['result']}")
        print("=" * 70)
        
        # 显示统计
        stats = self.get_statistics()
        print("\n运算统计:")
        for op, count in stats.items():
            print(f"  {op}: {count}次")

# 测试
history = CalculatorHistory()

history.add_record(10, '+', 5, 15)
history.add_record(20, '*', 3, 60)
history.add_record(100, '-', 25, 75)
history.add_record(50, '+', 30, 80)

history.show_all()

print("\n只看加法:")
for r in history.filter_by_operator('+'):
    print(f"{r['expression']} = {r['result']}")
```

"读取文件也有多种方式："

```python
# 方式1：read() - 一次读取全部
with open("data.txt", "r") as file:
    content = file.read()
    print(content)

# 方式2：readline() - 每次读一行
with open("data.txt", "r") as file:
    line1 = file.readline()
    line2 = file.readline()
    print(line1)
    print(line2)

# 方式3：readlines() - 读取所有行，返回列表
with open("data.txt", "r") as file:
    lines = file.readlines()
    for line in lines:
        print(line.strip())  # strip()去掉换行符

# 方式4：for循环遍历（推荐）
with open("data.txt", "r") as file:
    for line in file:
        print(line.strip())
```

```python
def save_history(history, filename="history.txt"):
    """保存历史记录到文件"""
    try:
        with open(filename, "w", encoding="utf-8") as file:
            for record in history:
                time_str = record['time'].strftime("%Y-%m-%d %H:%M:%S")
                line = f"{time_str}|{record['num1']}|{record['operator']}|{record['num2']}|{record['result']}\n"
                file.write(line)
        print(f"历史记录已保存到 {filename}")
        return True
    except Exception as e:
        print(f"保存失败: {e}")
        return False

def load_history(filename="history.txt"):
    """从文件加载历史记录"""
    try:
        from datetime import datetime
        history = []
        
        with open(filename, "r", encoding="utf-8") as file:
            for line in file:
                parts = line.strip().split('|')
                if len(parts) == 5:
                    record = {
                        'time': datetime.strptime(parts[0], "%Y-%m-%d %H:%M:%S"),
                        'num1': float(parts[1]),
                        'operator': parts[2],
                        'num2': float(parts[3]),
                        'result': float(parts[4])
                    }
                    history.append(record)
        
        print(f"成功加载 {len(history)} 条历史记录")
        return history
        
    except FileNotFoundError:
        print("历史文件不存在，将创建新文件")
        return []
    except Exception as e:
        print(f"加载失败: {e}")
        return []
```

```python
def save_history_json(history, filename="history.json"):

            }
            history.append(record)

    except FileNotFoundError:
```

```python
import json
from datetime import datetime
from pathlib import Path

class Calculator:
    """带文件持久化的计算器"""
    
    def __init__(self, data_dir="data"):
        self.data_dir = Path(data_dir)
        self.data_dir.mkdir(exist_ok=True)
        self.history_file = self.data_dir / "history.json"
        self.history = self.load_history()
    
    def save_history(self):
        """保存历史记录"""
        try:
            data = []
            for record in self.history:
                item = {
                    'time': record['time'].strftime("%Y-%m-%d %H:%M:%S"),
                    'num1': record['num1'],
                    'operator': record['operator'],
                    'num2': record['num2'],
                    'result': record['result']
                }
                data.append(item)
            
            with open(self.history_file, "w", encoding="utf-8") as f:
                json.dump(data, f, ensure_ascii=False, indent=2)
            
            return True
        except Exception as e:
            print(f"保存失败: {e}")
            return False
    
    def load_history(self):
        """加载历史记录"""
        try:
            if not self.history_file.exists():
                print("首次运行，创建新历史文件")
                return []
            
            with open(self.history_file, "r", encoding="utf-8") as f:
                data = json.load(f)
            
            history = []
            for item in data:
                record = {
                    'time': datetime.strptime(item['time'], "%Y-%m-%d %H:%M:%S"),
                    'num1': item['num1'],
                    'operator': item['operator'],
                    'num2': item['num2'],
                    'result': item['result']
                }
                history.append(record)
            
            print(f"加载了 {len(history)} 条历史记录")
            return history
            
        except Exception as e:
            print(f"加载失败: {e}，创建新历史")
            return []
    
    def add_record(self, num1, operator, num2, result):
        """添加记录"""
        record = {
            'time': datetime.now(),
            'num1': num1,
            'operator': operator,
            'num2': num2,
            'result': result
        }
        self.history.append(record)
    
    def show_history(self):
        """显示历史"""
        if not self.history:
            print("暂无历史记录")
            return
        
        print("\n" + "=" * 60)
        print("  计算历史")
        print("=" * 60)
        for i, record in enumerate(self.history, 1):
            time_str = record['time'].strftime("%Y-%m-%d %H:%M:%S")
            expr = f"{record['num1']} {record['operator']} {record['num2']}"
            print(f"{i}. [{time_str}] {expr} = {record['result']}")
        print("=" * 60)
    
    def cleanup(self):
        """清理资源，保存数据"""
        print("\n正在保存数据...")
        if self.save_history():
            print("数据已保存")
        else:
            print("数据保存失败")

# 使用
calc = Calculator()
calc.add_record(10, '+', 5, 15)
calc.add_record(20, '*', 3, 60)
calc.show_history()
calc.cleanup()
```

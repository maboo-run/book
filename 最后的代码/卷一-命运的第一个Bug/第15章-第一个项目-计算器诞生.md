```python
def calculate(num1, operator, num2):
    """
    执行基本四则运算
    
    参数:
        num1 (float): 第一个数字
        operator (str): 运算符 (+, -, *, /)
        num2 (float): 第二个数字
    
    返回:
        float: 计算结果
        None: 如果出错（比如除以零）
    """
    if operator == '+':
        return num1 + num2
    elif operator == '-':
        return num1 - num2
    elif operator == '*':
        return num1 * num2
    elif operator == '/':
        if num2 == 0:
            return None  # 除以零返回None
        return num1 / num2
    else:
        return None  # 不支持的运算符
```

```python
def calculate(num1, operator, num2):
    """核心计算函数"""
    if operator == '+':
        return num1 + num2
    elif operator == '-':
        return num1 - num2
    elif operator == '*':
        return num1 * num2
    elif operator == '/':
        if num2 == 0:
            return None
        return num1 / num2
    else:
        return None

def show_menu():
    """显示菜单"""
    print("\n" + "=" * 40)
    print("         欢迎使用超级计算器")
    print("=" * 40)
    print("支持的运算符: + - * /")
    print("输入 'q' 退出程序")

def get_number(prompt):
    """获取数字输入"""
    while True:
        user_input = input(prompt)
        if user_input.lower() == 'q':
            return None
        try:
            return float(user_input)
        except ValueError:
            print("输入错误！请输入有效的数字。")

def main():
    """主程序"""
    history = []  # 历史记录
    
    show_menu()
    
    while True:
        # 获取第一个数
        num1 = get_number("请输入第一个数(或输入q退出): ")
        if num1 is None:
            break
        
        # 获取运算符
        operator = input("请输入运算符(+ - * /): ")
        if operator.lower() == 'q':
            break
        
        # 获取第二个数
        num2 = get_number("请输入第二个数: ")
        if num2 is None:
            break
        
        # 计算

        if result is None:
            if operator == '/' and num2 == 0:

            history.append(f"{num1} {operator} {num2} = {result}")
    
    # 退出时显示历史
    if history:
        print("\n" + "=" * 40)
        print("         计算历史")
        print("=" * 40)
        for i, record in enumerate(history, 1):
            print(f"{i}. {record}")
        print("=" * 40)
    
    print("\n感谢使用！再见！")

if __name__ == "__main__":
```

```python
def main():
    """主程序"""
    history = []
    show_menu()
    
    previous_result = None  # 保存上次结果
    
    while True:
        # 如果有上次结果，询问是否继续
        if previous_result is not None:
            choice = input(f"\n使用上次结果 {previous_result} 继续计算? (y/n): ")
            if choice.lower() == 'y':
                num1 = previous_result
            else:
                num1 = get_number("请输入第一个数(或输入q退出): ")
                if num1 is None:
                    break
        else:
            num1 = get_number("请输入第一个数(或输入q退出): ")
            if num1 is None:
                break
        
        # 获取运算符
        operator = input("请输入运算符(+ - * /): ")
        if operator.lower() == 'q':
            break
        
        # 获取第二个数
        num2 = get_number("请输入第二个数: ")
        if num2 is None:
            break
        
        # 计算

        if result is None:
```

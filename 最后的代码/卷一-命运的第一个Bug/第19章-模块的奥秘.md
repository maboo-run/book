第一步，创建核心计算模块`core.py`：

```python
"""
核心计算模块
提供基本的四则运算功能
"""

def calculate(num1, operator, num2):
    """
    执行计算
    
    参数:
        num1: 第一个数
        operator: 运算符 (+, -, *, /, **)
        num2: 第二个数
    
    返回:
        tuple: (success, result_or_error)
    """
    try:
        if operator == '+':
            result = num1 + num2
        elif operator == '-':
            result = num1 - num2
        elif operator == '*':
            result = num1 * num2
        elif operator == '/':
            if num2 == 0:
                raise ZeroDivisionError("除数不能为零")
            result = num1 / num2
        elif operator == '**':
            result = num1 ** num2
        else:
            raise ValueError(f"不支持的运算符: {operator}")
        
        return (True, result)
    except Exception as e:
        return (False, str(e))
```

第二步，创建历史管理模块`history.py`：

```python
"""
历史记录管理模块
"""

from datetime import datetime

class HistoryManager:
    """历史记录管理器"""
    
    def __init__(self):
        self.records = []
    
    def add(self, num1, operator, num2, result):
        """添加记录"""
        record = {
            'time': datetime.now(),
            'num1': num1,
            'operator': operator,
            'num2': num2,
            'result': result
        }
        self.records.append(record)
    
    def get_all(self):
        """获取所有记录"""
        return self.records
    
    def clear(self):
        """清空历史"""
        self.records = []
    
    def show(self):
        """显示历史"""
        if not self.records:
            print("暂无历史记录")
            return
        
        print("\n" + "=" * 60)
        print("  计算历史")
        print("=" * 60)
        for i, record in enumerate(self.records, 1):
            time_str = record['time'].strftime("%H:%M:%S")
            expr = f"{record['num1']} {record['operator']} {record['num2']}"
            print(f"{i}. [{time_str}] {expr} = {record['result']}")
        print("=" * 60)
```

第三步，创建文件存储模块`storage.py`：

```python
"""
文件存储模块
负责历史记录的持久化
"""

import json
from datetime import datetime
from pathlib import Path

def save_to_file(records, filename="data/history.json"):
    """保存记录到文件"""
    try:
        filepath = Path(filename)
        filepath.parent.mkdir(parents=True, exist_ok=True)
        
        data = []
        for record in records:
            item = {
                'time': record['time'].strftime("%Y-%m-%d %H:%M:%S"),
                'num1': record['num1'],
                'operator': record['operator'],
                'num2': record['num2'],
                'result': record['result']
            }
            data.append(item)
        
        with open(filepath, 'w', encoding='utf-8') as f:
            json.dump(data, f, ensure_ascii=False, indent=2)
        
        return True
    except Exception as e:
        print(f"保存失败: {e}")
        return False

def load_from_file(filename="data/history.json"):
    """从文件加载记录"""
    try:
        filepath = Path(filename)
        if not filepath.exists():
            return []
        
        with open(filepath, 'r', encoding='utf-8') as f:
            data = json.load(f)
        
        records = []
        for item in data:
            record = {
                'time': datetime.strptime(item['time'], "%Y-%m-%d %H:%M:%S"),
                'num1': item['num1'],
                'operator': item['operator'],
                'num2': item['num2'],
                'result': item['result']
            }
            records.append(record)
        
        return records
    except Exception as e:
        print(f"加载失败: {e}")
        return []
```

```python
"""
用户界面模块
"""

def show_menu():
    """显示主菜单"""
    print("\n" + "=" * 60)
    print("  超级计算器 v2.0")
    print("=" * 60)
    print("  支持运算: + - * / **")
    print("  输入 'q' 退出 | 'h' 查看历史 | 'c' 清空历史")
    print("=" * 60 + "\n")

def get_number(prompt):
    """获取用户输入的数字"""
    while True:
        try:
            user_input = input(prompt)
            if user_input.lower() in ['q', 'h', 'c']:
                return user_input.lower()
            return float(user_input)
        except ValueError:
            print("输入错误！请输入有效的数字")
        except KeyboardInterrupt:
            return 'q'
```

第五步，创建主程序`main.py`：

```python
"""
主程序
整合所有模块
"""

from core import calculate
from history import HistoryManager
from storage import save_to_file, load_from_file
from ui import show_menu, get_number

def main():
    """主函数"""
    # 初始化
    history = HistoryManager()
    
    # 加载历史
    loaded_records = load_from_file()
    for record in loaded_records:
        history.records.append(record)
    
    if loaded_records:
        print(f"加载了 {len(loaded_records)} 条历史记录")
    
    show_menu()
    
    previous_result = None
    
    try:
        while True:
            # 获取第一个数
            if previous_result is not None:
                choice = input(f"使用上次结果 {previous_result:.2f} ? (y/n): ")
                if choice.lower() == 'y':
                    num1 = previous_result
                else:
                    num1 = get_number("请输入第一个数: ")
            else:
                num1 = get_number("请输入第一个数: ")
            
            # 处理特殊命令
            if num1 == 'q':
                break
            elif num1 == 'h':
                history.show()
                continue
            elif num1 == 'c':
                history.clear()
                print("历史记录已清空")
                continue
            
            # 获取运算符
            operator = input("请输入运算符(+ - * / **): ")
            if operator.lower() == 'q':
                break
            
            # 获取第二个数
            num2 = get_number("请输入第二个数: ")
            if num2 == 'q':
                break
            
            # 计算
            success, result = calculate(num1, operator, num2)
            
            if success:
                print(f"\n✓ 结果: {num1} {operator} {num2} = {result}\n")
                history.add(num1, operator, num2, result)
                previous_result = result
            else:
                print(f"\n✗ 错误: {result}\n")
                previous_result = None
        
        # 保存历史
        print("\n正在保存数据...")
        if save_to_file(history.get_all()):
            print("数据已保存")
        
        print("感谢使用！")
    
    except Exception as e:
        print(f"\n程序异常: {e}")
        # 尝试保存数据
        save_to_file(history.get_all())

if __name__ == "__main__":
    main()
```

```python
# test.py
print(f"__name__ = {__name__}")

def hello():
    print("Hello!")

if __name__ == "__main__":
    print("这个文件是直接运行的")
    hello()
```


在另一个文件中import：

```python
import test  # 只会输出: __name__ = test

```

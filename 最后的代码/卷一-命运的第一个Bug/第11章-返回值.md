```python
# 错误的做法:只打印,不返回
def calculate_discount_wrong(price, discount):
    final_price = price * discount
    print(f"折后价: {final_price}")

# 正确的做法:返回结果
def calculate_discount(price, discount):
    final_price = price * discount
    return final_price

result1 = calculate_discount_wrong(100, 0.8)

print(f"result2 = {result2}")  # 80.0
```

```python
def is_prime(num):
    if num < 2:
        return False
    if num == 2:
        return True
    if num % 2 == 0:
        return False
    
    # 只需检查到sqrt(num)
    i = 3
    while i * i <= num:
        if num % i == 0:
            return False
        i += 2
    
    return True
```

```python
def calculate_bmi(height, weight):
    """
    计算BMI指数
    
    参数:
        height: 身高(米)
        weight: 体重(公斤)
    
    返回:
        BMI值(float)
    """
    return weight / (height ** 2)

def get_bmi_status(bmi):
    """
    根据BMI判断体重状态
    
    参数:
        bmi: BMI指数
    
    返回:
        状态字符串
    """
    if bmi < 18.5:
        return "偏瘦"
    elif bmi < 24:
        return "正常"
    elif bmi < 28:
        return "偏胖"
    else:
        return "肥胖"

# 主程序
height = float(input("请输入身高(米): "))
weight = float(input("请输入体重(公斤): "))

bmi = calculate_bmi(height, weight)
status = get_bmi_status(bmi)

print(f"你的BMI是: {bmi:.2f}")
print(f"体重状况: {status}")
```
